// ------------ IPA ------------



# Build (no signing)
xcodebuild \
  -project Shirox.xcodeproj \
  -scheme Shirox_iOS \
  -sdk iphoneos \
  -destination 'generic/platform=iOS' \
  -configuration Release \
  CODE_SIGN_IDENTITY="" CODE_SIGNING_REQUIRED=NO CODE_SIGNING_ALLOWED=NO \
  CONFIGURATION_BUILD_DIR=build/app \
  build

# Create IPA with correct permissions
rm -rf build/Payload
mkdir -p build/Payload
ditto build/app/Shirox.app build/Payload/Shirox.app
chmod -R 755 build/Payload/Shirox.app
chmod +x build/Payload/Shirox.app/Shirox

cd build && zip -r Shirox.ipa Payload/ && cd ..
